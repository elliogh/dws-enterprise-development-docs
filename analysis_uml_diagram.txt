class Book {
  - id: Long
  - title: String
  - author: String
  - description: String
  - price: BigDecimal
  - isAvailable: Boolean
}

class Cart {
  - id: UUID
  - items: List<CartItem>
}

class CartItem {
  - book: Book
  - quantity: Integer
}

Cart --> CartItem : contains
CartItem --> Book : association
